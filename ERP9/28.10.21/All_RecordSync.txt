[Function: Func_Sync_All_Data_To_TrackPayout]
	
	10	: Msg Box	: "Hello"		: "TrackPayout"
	
	20	: Call		: Func_Sync_All_Ledger
	22	: If		: @@LedgerPostStatus = 200
	;23	: Display	: Test1
	24	: Call		: Func_Sync_All_SalesVoucher
	45	: Call		: Func_Sync_All_ReceiptVoucher
		
		;45f	: Display	: Test2	
		;45h	: Display	: Test3	

		45a	: If: @@SalesPostStatus = 200 and @@ReceiptPostStatus = 200
		45d	: Call	:  Func UpdateID
		45e: Else
		45f: Msg Box	: "Status" : "Error Code" + $$NewLine + "Sale Failed : " + @@SalesPostStatus + $$NewLine + "Receipt Failed : " + @@ReceiptPostStatus
		45m	: End If

	45o	: Else
	45p	: Msg Box	: "Status" : "Error Code" + $$NewLine +   @@LedgerPostStatus
	50	: End If

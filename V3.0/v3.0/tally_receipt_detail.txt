	Set		: SVToDate	: $$Date:$LASTVOUCHERDATE:Company:$$CurrentCompany

	Part: TPO_Receipt_ColDetail
	
	Height	: 100 % Page
	Width	: 100 % Page
	
[Part: TPO_Receipt_ColDetail]
	
	Line	: TPO_Receipt_ColDetail
	Repeat	: TPO_Receipt_ColDetail	: TPO_Receipt_LedgerEntries_Collection_Src
	Scroll	: Vertical	
	
[Collection: TPO_Receipt_Collection_Src]
	
	Parm Var	: SVFromDate	: Date: $StartingFrom:Company:$$CurrentCompany
	Parm Var	: SVToDate		: Date: $LASTVOUCHERDATE:Company:$$CurrentCompany
	
	Type		: Vouchers: VoucherType
	Child Of	: $$VchTypeReceipt
	Belongs To	: Yes
	Fetch		: ?,alterid
	Fetch		: ledgerEntries.*, BankAllocations.*
	Filter		: GetFiltertheReport
	
[Collection: TPO_Receipt_LedgerEntries_Collection_Src]
	
	Source Collection	: TPO_Receipt_Collection_Src
	Walk				: LedgerEntries
	Fetch		: ?,Guid,VoucherTypeName, PartyLedgerName, VoucherNumber, Date, IsOptional, IsCancelled, isDeleted, IsPostDated

	Field: TPO_Receipt_subscription_id, TPO_Receipt_company_id,	TPO_Receipt_guid, TPO_Receipt_voucher_type, TPO_Receipt_voucher_type_alias
	Field: TPO_Receipt_date	, TPO_Receipt_voucher_number, TPO_Receipt_ledger_id
	Field: TPO_Receipt_is_optional, TPO_Receipt_is_deleted, TPO_Receipt_is_post_dated, TPO_Receipt_is_cancelled
	Field: TPO_Receipt_amount, TPO_Receipt_is_deemed_positive
	
	JSONTag	:"Receipt"

	Local	: Field: Default	: Type	: String
	Local	: Field: Default	: Border	: THick Left
	
	
	Explode	: TPO_Receipt_Bill_Allocations: Yes
	
[Part: TPO_Receipt_Bill_Allocations]
	
	Line	: TPO_Receipt_Bill_Allocations
	Repeat	: TPO_Receipt_Bill_Allocations		: BillAllocations
	Scroll	: Vertical
	
[Line: TPO_Receipt_Bill_Allocations]
		
	Field: TPO_Receipt_bill_type,TPO_Receipt_bill_date
	Field: TPO_Receipt_bill_name, TPO_Receipt_bill_due_date, TPO_Receipt_bill_amount, TPO_Receipt_bill_dr_or_cr

	JSONTag		: "bill_allocation"
	
[Field: TPO_Receipt_bill_type]
	
	
	Set as	: $BillType
	JSONTag	: "bill_type"
	
[Field: TPO_Receipt_bill_date]
	
	Set as	: if $$IsEmpty:$BillDate then $$TPO_DateMMYYYYFormat:$$Date:$BillDate else $$TPO_DateMMYYYYFormat:$$Date:$BillDate
	JSONTag	: "bill_date"
	
[Field:  TPO_Receipt_bill_name]
	
	Set as	: if $$IsEmpty:$Name then "NA" else $Name
	JSONTag	: "bill_name"
	
[Field: TPO_Receipt_bill_due_date]

	Type	: Due Date
	Set as	: if $$IsEmpty:$BillCreditPeriod then $$TPO_DateMMYYYYFormat:$$Date:$BillDate else $$TPO_DateMMYYYYFormat:$$Date:$BillCreditPeriod
	JSONTag	: "bill_due_date"
	
[Field: TPO_Receipt_bill_amount]
		
	Set as	: $$TPO_RemoveCharactersFromString:$amount:","
	JSONTag	: "bill_amount"
	
[Field: TPO_Receipt_bill_dr_or_cr]
		
	Set as	: if $$IsDr:$amount then "Yes" else "No"; $IsDeemedPositive
	JSONTag	: "is_deemed_positive"
	
[Field: TPO_Receipt_subscription_id]
	
	Set as	: @@TPO_CMPSubscriptionDetail
	JSONTag	: "subscription_id"
[Field: TPO_Receipt_company_id]
	
	Set as	: @@IsTrackPayout_CMPID
	JSONTag	: "company_id"
	
[Field: TPO_Receipt_guid]

	Set as	: $Guid
	JSONTag	: "guid"
[Field: TPO_Receipt_voucher_type]
	
	Set as	: $voucherTypeName
	JSONTag	: "voucher_type"
[Field: TPO_Receipt_voucher_type_alias]
	
	Set as	: "Receipt"
	JSONTag	: "voucher_type_alias"
[Field: TPO_Receipt_date]
	
	Set as	: $$TPO_DateMMYYYYFormat:$$Date:$Date
	JSONTag	: "date"
[Field: TPO_Receipt_voucher_number]
	
	Set as	: $VoucherNUmber
	JSONTag	: "voucher_number"
[Field: TPO_Receipt_ledger_id]
	
	Set as	: $guid:Ledger:$LedgerName
	JSONTag: "ledger_id"
[Field: TPO_Receipt_is_optional]
	
	Set as	: $IsOptional
	JSONTag	: "is_optional"
	
[Field: TPO_Receipt_is_deleted]
		
	Set as	: $IsDeleted
	JSONTag	: "is_deleted"
[Field: TPO_Receipt_is_post_dated]
		
	Set as	: $IsPostDated
	JSONTag	: "is_post_dated"
	
[Field: TPO_Receipt_is_cancelled]
	
	Set as	: $IsCancelled
	JSONTag	: "is_cancelled"
	
[Field: TPO_Receipt_amount]
	
	Set as	: $$TPO_RemoveCharactersFromString:$Amount:","
	JSONTag	: "amount"
	
[Field: TPO_Receipt_is_deemed_positive]

	Set as	: if $$IsDr:$amount = "Yes" then "Yes" else "No"
	JSONTag	: "is_deemed_positive"
